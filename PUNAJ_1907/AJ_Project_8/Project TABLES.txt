create table  doctor_details
(
 fname varchar2(15)not null,
 lname varchar2(15),
 age number(3),
 contact_no number(10),
 reg_id varchar2(25),
 gender varchar2(10),
 city varchar2(15),
 state varchar2(20),
 password varchar2(20),
 user_id varchar2(10) Primary key,
SECURITY_QUESTION varchar2(30),
SECURITY_ANSWER varchar2(30)
);
